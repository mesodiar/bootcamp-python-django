def new_story(request):
    if request.method == "POST":
        form = StoryForm(request.POST, request.FILES)
        if form.is_valid():
            form.save()

            return redirect('index')
    else:
        form = StoryForm()
    return render(request, 'stories/new_story.html', {'form': form})
